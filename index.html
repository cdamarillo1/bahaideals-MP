<script>

const SCRIPT_URL="https://script.google.com/macros/s/AKfycbz5AJnYufkOZQ8fsDk03e07A_sJ61e8NgK30dN0CJq3mQ4gKA0F4BTu5RK3c6ko1UAf/exec";

let allMPs=[],allLeads=[],activeFilter="ALL";

function ini(n){return n.split(" ").slice(0,2).map(w=>w[0]).join("").toUpperCase()}

function tagCls(t){return t==="customer"?"pill-customer":t==="mp"?"pill-mp":"pill-signup"}

function leadsFor(tag){return allLeads.filter(l=>(l.mp_tag||"").toUpperCase().trim()===tag.toUpperCase())}

function toggle(id){document.getElementById("b_"+id).classList.toggle("open");document.getElementById("c_"+id).classList.toggle("open")}

function loadData(){

  document.getElementById("error").style.display="none";

  var cb="cb_"+Date.now();

  var script=document.createElement("script");

  var timeout=setTimeout(function(){

    document.getElementById("error").textContent="Could not load data.";

    document.getElementById("error").style.display="block";

    document.getElementById("mpList").innerHTML="";

    delete window[cb];

  },8000);

  window[cb]=function(json){

    clearTimeout(timeout);

    script.remove();

    delete window[cb];

    if(!json.success)return;

    allMPs=json.mps;

    allLeads=json.leads;

    document.getElementById("lastRefresh").textContent="Updated "+new Date().toLocaleTimeString();

    render();

  };

  script.src=SCRIPT_URL+"?callback="+cb;

  document.head.appendChild(script);

}

function render(){

  var search=document.getElementById("search").value.toLowerCase();

  var withLeads=allMPs.filter(function(mp){return leadsFor(mp.tag).length>0});

  document.getElementById("statGrid").innerHTML='<div class="stat"><div class="stat-label">Total MPs</div><div class="stat-val">'+allMPs.length+'</div></div><div class="stat"><div class="stat-label">With Leads</div><div class="stat-val">'+withLeads.length+'</div></div><div class="stat"><div class="stat-label">Total Leads</div><div class="stat-val">'+allLeads.length+'</div></div><div class="stat"><div class="stat-label">No Leads</div><div class="stat-val">'+(allMPs.length-withLeads.length)+'</div></div>';

  document.getElementById("filterRow").innerHTML=[{k:"ALL",label:"All MPs"},{k:"WITH",label:"Has leads"},{k:"WITHOUT",label:"No leads yet"}].map(function(f){return'<div class="filter-btn'+(activeFilter===f.k?" active":"")+'" onclick="activeFilter=\''+f.k+'\';render()">'+f.label+'</div>'}).join("");

  var mps=allMPs.filter(function(mp){if(activeFilter==="WITH")return leadsFor(mp.tag).length>0;if(activeFilter==="WITHOUT")return leadsFor(mp.tag).length===0;return true});

  if(search)mps=mps.filter(function(mp){return mp.name.toLowerCase().includes(search)||(mp.email||"").toLowerCase().includes(search)||leadsFor(mp.tag).some(function(l){return(l.lead_name||"").toLowerCase().includes(search)})});

  mps.sort(function(a,b){return leadsFor(b.tag).length-leadsFor(a.tag).length});

  var html="";

  mps.forEach(function(mp){

    var leads=leadsFor(mp.tag),has=leads.length>0,sid=mp.tag.replace(/[^a-z0-9]/gi,"_");

    html+='<div class="mp-block"><div class="mp-header" onclick="toggle(\''+sid+'\')"><div class="av'+(has?" av-has":"")+'">'+ini(mp.name)+'</div><div class="mp-info-wrap"><div class="mp-name">'+mp.name+'</div><div class="mp-email">'+mp.email+'</div></div><span class="cnt'+(has?"":" cnt-none")+'">'+(has?leads.length+" lead"+(leads.length!==1?"s":""):"no leads")+'</span><span class="chev" id="c_'+sid+'">&#9658;</span></div><div class="mp-body" id="b_'+sid+'"><div class="mp-contact">'+(mp.phone?"<span>"+mp.phone+"</span>":"")+"<span>"+mp.email+"</span></div>";

    if(has){leads.forEach(function(l){html+='<div class="lead-row"><div class="lead-dot">'+ini(l.lead_name||"")+'</div><div style="flex:1;min-width:0"><div class="lead-name">'+l.lead_name+'</div><div class="lead-meta"><span class="pill '+tagCls(l.tag)+'">'+l.tag+'</span><span class="pill pill-stage">'+l.stage+'</span><span class="date-txt">'+l.date+'</span>'+(l.notes?'<span class="date-txt">· '+l.notes+"</span>":"")+"</div></div></div>"})}

    else{html+='<div class="no-leads">No leads assigned yet</div>'}

    html+="</div></div>";

  });

  document.getElementById("mpList").innerHTML=html||'<div class="loading">No results found.</div>';

}

loadData();

setInterval(loadData,60000);

</script>
